# user_diskio

## Overview

This module provides an **SPI SD-card interface** used to initialize the card and perform block-level read/write operations.
It is intended to implement the **STM32CubeIDE-generated FATFS "User-defined" disk I/O layer**,
typically in `user_diskio.c`.

In practice, this file acts as the bridge between:

- **FatFs / diskio API** generated by STM32CubeIDE
- the low-level **SD over SPI driver** (`sd_spi.c/.h`)
- the physical SD card connected to the MCU

## Purpose

The purpose of this module is to expose the standard FatFs disk I/O callbacks required by the middleware:

- `USER_initialize()`
- `USER_status()`
- `USER_read()`
- `USER_write()`
- `USER_ioctl()`

These functions translate FatFs requests into block-level SD card operations through the custom SPI SD driver.

## Notes

If you have a `.c` file inside a `doc/` folder that is only meant for documentation, it should **not** be added to the build source list.
In STM32CubeIDE, the cleanest approach is:

- Right click the file
- `Resource Configurations -> Exclude from Build...`
- Mark `Debug` and `Release`

That keeps the file inside the project without compiling it.

/* USER CODE BEGIN DECL */

/* Includes ------------------------------------------------------------------*/
#include "ff_gen_drv.h"
#include "sd_spi.h"

/* Private typedef -----------------------------------------------------------*/
/* Private define ------------------------------------------------------------*/

/* Private variables ---------------------------------------------------------*/
/* Disk status */
static volatile DSTATUS s_stat = STA_NOINIT;
extern SPI_HandleTypeDef SD_CARD_SPI;


static bool user_drive_is_valid(BYTE pdrv)
{
	return (pdrv == SD_SPI_PHYSICAL_DRIVE);
}

static DRESULT sd_spi_status_to_dresult(sd_spi_status_t st)
{
	switch (st)
	{
		case SD_SPI_OK:
			return RES_OK;

		case SD_SPI_NOT_READY:
			return RES_NOTRDY;

		case SD_SPI_TIMEOUT:
		case SD_SPI_ERR:
		default:
			return RES_ERROR;
	}
}

/* USER CODE END DECL */

/* Private function prototypes -----------------------------------------------*/
DSTATUS USER_initialize (BYTE pdrv);
DSTATUS USER_status (BYTE pdrv);
DRESULT USER_read (BYTE pdrv, BYTE *buff, DWORD sector, UINT count);
#if _USE_WRITE == 1
  DRESULT USER_write (BYTE pdrv, const BYTE *buff, DWORD sector, UINT count);
#endif /* _USE_WRITE == 1 */
#if _USE_IOCTL == 1
  DRESULT USER_ioctl (BYTE pdrv, BYTE cmd, void *buff);
#endif /* _USE_IOCTL == 1 */

Diskio_drvTypeDef  USER_Driver =
{
  USER_initialize,
  USER_status,
  USER_read,
#if  _USE_WRITE
  USER_write,
#endif  /* _USE_WRITE == 1 */
#if  _USE_IOCTL == 1
  USER_ioctl,
#endif /* _USE_IOCTL == 1 */
};

/* Private functions ---------------------------------------------------------*/

/**
  * @brief  Initializes a Drive
  * @param  pdrv: Physical drive number (0..)
  * @retval DSTATUS: Operation status
  */
DSTATUS USER_initialize (
	BYTE pdrv           /* Physical drive nmuber to identify the drive */
)
{
  /* USER CODE BEGIN INIT */
	if (!user_drive_is_valid(pdrv))
	{
		s_stat = STA_NOINIT;

		return s_stat;
	}

	if (sd_spi_is_ready())
	{
		s_stat = 0u;

		return s_stat;
	}

	sd_spi_status_t st = sd_spi_init();
	if (st == SD_SPI_OK)
	{
		s_stat = 0u;
	}
	else
	{
		s_stat = STA_NOINIT;
	}

	return s_stat;
  /* USER CODE END INIT */
}

/**
  * @brief  Gets Disk Status
  * @param  pdrv: Physical drive number (0..)
  * @retval DSTATUS: Operation status
  */
DSTATUS USER_status (
	BYTE pdrv       /* Physical drive number to identify the drive */
)
{
  /* USER CODE BEGIN STATUS */
	if (!user_drive_is_valid(pdrv))
	{
		s_stat = STA_NOINIT;

		return s_stat;
	}

	if (sd_spi_is_ready())
	{
		s_stat = 0u;
	}
	else
	{
		s_stat = STA_NOINIT;
	}

	return s_stat;
  /* USER CODE END STATUS */
}

/**
  * @brief  Reads Sector(s)
  * @param  pdrv: Physical drive number (0..)
  * @param  *buff: Data buffer to store read data
  * @param  sector: Sector address (LBA)
  * @param  count: Number of sectors to read (1..128)
  * @retval DRESULT: Operation result
  */
DRESULT USER_read (
	BYTE pdrv,      /* Physical drive nmuber to identify the drive */
	BYTE *buff,     /* Data buffer to store read data */
	DWORD sector,   /* Sector address in LBA */
	UINT count      /* Number of sectors to read */
)
{
  /* USER CODE BEGIN READ */
	if (!user_drive_is_valid(pdrv))
	{
		return RES_PARERR;
	}

	if ((buff == NULL) || (count == 0u))
	{
		return RES_PARERR;
	}

	if ((USER_status(pdrv) & STA_NOINIT) != 0u)
	{
		return RES_NOTRDY;
	}

	sd_spi_status_t st = sd_spi_read_blocks(buff, (uint32_t)sector, (uint32_t)count);

	return sd_spi_status_to_dresult(st);
  /* USER CODE END READ */
}

/**
  * @brief  Writes Sector(s)
  * @param  pdrv: Physical drive number (0..)
  * @param  *buff: Data to be written
  * @param  sector: Sector address (LBA)
  * @param  count: Number of sectors to write (1..128)
  * @retval DRESULT: Operation result
  */
#if _USE_WRITE == 1
DRESULT USER_write (
	BYTE pdrv,          /* Physical drive nmuber to identify the drive */
	const BYTE *buff,   /* Data to be written */
	DWORD sector,       /* Sector address in LBA */
	UINT count          /* Number of sectors to write */
)
{
  /* USER CODE BEGIN WRITE */
	if (!user_drive_is_valid(pdrv))
	{
		return RES_PARERR;
	}

	if ((buff == NULL) || (count == 0u))
	{
		return RES_PARERR;
	}

	if ((USER_status(pdrv) & STA_NOINIT) != 0u)
	{
		return RES_NOTRDY;
	}

	sd_spi_status_t st = sd_spi_write_blocks(buff, (uint32_t)sector, (uint32_t)count);

	return sd_spi_status_to_dresult(st);
  /* USER CODE END WRITE */
}
#endif /* _USE_WRITE == 1 */

/**
  * @brief  I/O control operation
  * @param  pdrv: Physical drive number (0..)
  * @param  cmd: Control code
  * @param  *buff: Buffer to send/receive control data
  * @retval DRESULT: Operation result
  */
#if _USE_IOCTL == 1
DRESULT USER_ioctl (
	BYTE pdrv,      /* Physical drive nmuber (0..) */
	BYTE cmd,       /* Control code */
	void *buff      /* Buffer to send/receive control data */
)
{
  /* USER CODE BEGIN IOCTL */
	if (!user_drive_is_valid(pdrv))
	{
		return RES_PARERR;
	}

	switch (cmd)
	{
		case CTRL_SYNC:
		{
			if ((USER_status(pdrv) & STA_NOINIT) != 0u)
			{
				return RES_NOTRDY;
			}

			return RES_OK;
		}

		case GET_SECTOR_COUNT:
		{
			if (buff == NULL)
			{
				return RES_PARERR;
			}

			if ((USER_status(pdrv) & STA_NOINIT) != 0u)
			{
				return RES_NOTRDY;
			}

			sd_spi_status_t st = sd_spi_get_sector_count((uint32_t*)buff);

			return sd_spi_status_to_dresult(st);
		}

		case GET_SECTOR_SIZE:
		{
			if (buff == NULL)
			{
				return RES_PARERR;
			}

			*(WORD*)buff = (WORD)SD_SPI_SECTOR_SIZE;

			return RES_OK;
		}

		case GET_BLOCK_SIZE:
		{
			if (buff == NULL)
			{
				return RES_PARERR;
			}

			*(DWORD*)buff = (DWORD)SD_SPI_ERASE_BLOCK_SIZE;

			return RES_OK;
		}

		default:
			return RES_PARERR;
	}
  /* USER CODE END IOCTL */
}
#endif /* _USE_IOCTL == 1 */
